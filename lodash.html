Collections

Every


_.every([1, 2, 3, 4], function(n) {
	return n % 2 === 0;
});
// -> false

var fruits = [
{
	name: 'apple',
	price: 1.99,
	onSale: true
},
{
	name: 'orange',
	price: 0.99,
	onSale: true
}
];
_.every(fruits, 'onSale', true);
// -> true

var obj = {
	a: 1,
	b: 2,
	c: 3
};
_.every(obj, function(n) {
	return n % 2 === 0;
});
// -> false

Collections

Filter


_.filter(['a', 'b', 'c'], function(char) {
	return char > 'b';
});
// -> ['c']

var obj = {
	a: 1,
	b: 2,
	c: 3
};
_.filter(obj, function(n) {
	return n > 1;
});
// -> [2, 3]

_.filter('hello', function(c) {
	return c !== 'l';
});
// -> ['h', 'e', 'o']

_.reject(['a', 'b', 'c'], function(char) {
	return char > 'b';
});
// -> ['a', 'b']

Count by


_.countBy([1, 2, 3], function(n) {
	return n > 1;
});
// -> {false: 1, true: 2}

	a: 1,

});

	return char === 'l';
});
// -> {false: 3, true: 2}

Collection

Group by 


_.groupBy([1, 2, 3], function(n) {
	return n > 1;
});
// -> {false: [1], true: [2, 3]}

_.groupBy({
	a: 1,
	b: 1,
	c: 2
}, function(val) {
	return val / 2;
});
// -> {0.5: [1, 1], 1: [2]}

_.groupBy('hello', function(char) {
	return char === 'l';
});
// -> {fasle: ['h', 'e', o], true: ['l', 'l']}

Object

keys, values


function Foo() {
	this.a = 1;
	this.b = 2;
}

Foo.prototype.c = 3;

_.keys(new Foo);
// → ['a', 'b'] (iteration order is not guaranteed)

_.keys('hi');
// → ['0', '1']

function Foo() {
	this.a = 1;
	this.b = 2;
}

Foo.prototype.c = 3;

_.values(new Foo);
// → [1, 2] (iteration order is not guaranteed)

_.values('hi');
// → ['h', 'i']

Lang

isFunction, isInteger, isObject, isArray



_.isFunction(_);
// → true

_.isFunction(/abc/);
// → false

_.isInteger(3);
// → true

_.isInteger(Number.MIN_VALUE);
// → false

_.isInteger(Infinity);
// → false

_.isInteger('3');
// → false

_.isObject({});
// → true

_.isObject([1, 2, 3]);
// → true

_.isObject(_.noop);
// → true

_.isObject(null);
// → false

_.isArray([1, 2, 3]);
// → true

_.isArray(document.body.children);
// → false

_.isArray('abc');
// → false

_.isArray(_.noop);
// → false
